def make_patient_visit_summary(text, patient_info, print_format):
    """
    Make the patient visit summary for printing.

    Parameters:
        text (str): The original text for the patient visit. This is the text
        generated by a visit summary.

        patient_info (dict): A dictionary with information about the patient.
        This contains information about the patient, as well as the visit,
        which the summary text needs for reference.

        print_format (str): A string with one or more options for how the
        summary will print.

    Returns:
        text (str): A string containing all of the data necessary for the
        patient visit summary, in a simple format for printing.
    """
    if print_format.upper() == "TEXT ONLY":
        print_format = "TEXT"
    elif print_format.upper() == "PDF":
        text_pdf = pdf_generator.make_pdf(text, patient_info["Record Date"], patient_info["Patient Name"])
        return text_pdf
    elif print_format.upper() == "HTML":
        text_html = html_generator.make_html(text, patient_info["Record Date"], patient_info["Patient Name"])
        return text_html
    elif print_format.upper() == "CHART":
        print_format = "CHART"
    elif print_format.upper() == "TEXT AND CHART":
        print_format = "TEXT AND CHART"
    else:
        print_format = "TEXT"
    return make_text_for_patient_chart_visit_summary(text, patient_info)
